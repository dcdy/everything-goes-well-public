获取系统设备信息和网关 ​

js
// 导入os模块获取设备信息
import * as os from 'os';
// 获取设备信息
ipcMain.handle('get_system_info', async () => {
    let info = {
        hostname: os.hostname(), //返回操作系统的主机名。
        type: os.type(), //返回操作系统名
        platform: os.platform(), //返回操作系统名
        arch: os.arch(), //返回操作系统 CPU 架构，可能的值有 "x64"、"arm" 和 "ia32"。
        release: os.release(), //返回操作系统的发行版本。
        uptime: os.uptime(), //返回操作系统运行的时间，以秒为单位。
        loadavg: os.loadavg(), //返回一个包含 1、5、15 分钟平均负载的数组。
        totalmem: os.totalmem(), //返回系统内存总量，单位为字节。
        freemem: os.freemem(), //返回操作系统空闲内存量，单位是字节。
        cpus: os.cpus(), //返回一个对象数组，包含所安装的每个 CPU/内核的信息：型号、速度（单位 MHz）、时间（一个包含 user、nice、sys、idle 和 irq 所使用 CPU/内核毫秒数的对象）。
        userInfo: os.userInfo(), //返回当前用户的信息
        tmpdir: os.tmpdir(), //返回操作系统的默认临时文件夹。
        net: os.networkInterfaces() //获得网络接口列表。
    };

    const interfaces = os.networkInterfaces();
    // 遍历网络接口
    Object.keys(interfaces).forEach((interfaceName) => {
        const interfaceInfo = interfaces[interfaceName];

        // 遍历每个接口的 IP 地址
        interfaceInfo?.forEach((detail) => {
            if (detail.family === 'IPv4' && !detail.internal && detail.address && detail.netmask) {
                // 忽略内部（非公共可访问）IP地址
                let netInfo = {
                    ipname: interfaceName, // 网络接口名称
                    address: detail.address, // IP地址
                    netmask: detail.netmask, // 子网掩码
                    mac: detail.mac, // MAC地址
                    cidr: detail.cidr // CIDR表示法
                };
                info = { ...info, ...netInfo };
            }
        });
    });

    // 获取默认网关
    info = { ...info, ...(await getDefaultGateway()) };

    return info;
});

// 获取默认网关的函数
function getDefaultGateway() {
    return new Promise((resolve, reject) => {
        exec('ipconfig', { encoding: 'buffer' }, function (error, stdout, stderr) {
            let gateway = '';
            let gatewayObj = {};

            if (error) {
                let newErr = iconv.decode(JSON.stringify(error), 'gbk');
                console.log('error:', newErr, typeof newErr);
                gateway = null;
                gatewayObj = {
                    gateway,
                    gatewayMsg: '获取默认网关执行出错'
                };
                return resolve({ gateway, gatewayObj });
                // return reject(error);
            }

            const ipconfigInfo = iconv.decode(stdout, 'gbk');
            const ipconfigInfoArr = ipconfigInfo.split('\n');

            let findWlan = false;

            ipconfigInfoArr.some((line, index) => {
                if (line.includes('WLAN')) {
                    findWlan = true;
                }

                if (findWlan && line.includes('默认网关. . . . . . . . . . . . .')) {
                    gateway = ipconfigInfoArr[index + 1].trim();
                    return true;
                }
            });

            if (!gateway) {
                console.log('找不到默认网关地址');
                gateway = null;
                gatewayObj = {
                    gateway,
                    gatewayMsg: '找不到默认网关地址',
                    ipconfigInfo,
                    ipconfigInfoArr
                };
                return resolve({ gateway, gatewayObj });
            }

            gatewayObj = {
                gateway,
                gatewayMsg: '成功',
                ipconfigInfo,
                ipconfigInfoArr
            };
            resolve({ gateway, gatewayObj });
        });
    });
}

// 更改主机名
function changeHostname(newHostname) {
    return new Promise((resolve, reject) => {
        // const newHostname = 'DESKTOP-E5KV2P8'; // 将 'newHostname' 替换成你想设置的新主机名
        let res = {};
        // let command = `powershell -Command "Rename-Computer -NewName '${newHostname}' -Force -Restart"`;
        let command = `powershell -Command "Rename-Computer -NewName '${newHostname}' -Force"`;
        exec(command, { shell: true, encoding: 'buffer' }, (error, stdout, stderr) => {
            if (error) {
                res = {
                    status: '失败',
                    msg: '修改电脑名称失败'
                };
                return resolve(res);
            }
            if (stderr) {
                let msg = iconv.decode(stdout, 'gbk');
                res = {
                    status: '成功',
                    msg
                };
                return resolve(res);
            }
        });
    });
}
ipcMain.handle('change_hostname', async (event, newHostname) => {
    console.log('newHostname', newHostname);
    let res = { ...(await changeHostname(newHostname)) };
    return res;
});

// 更改ip地址，子网掩码，网关
function changeNetInfo(netInfo) {
    return new Promise((resolve, reject) => {
        let res = {};
        const { address, netmask, gateway, ipname } = netInfo;
        exec(
            `netsh interface ip set address ${
                ipname || 'WLAN'
            } static ${address} ${netmask} ${gateway}`,
            (error, stdout, stderr) => {
                console.log('error', error, stdout, stderr);
                if (error) {
                    res = {
                        status: '失败',
                        msg: '修改ip地址，子网掩码，网关失败'
                    };
                    return resolve(res);
                }

                // let msg = iconv.decode(stdout, 'gbk');
                let msg = '修改ip地址，子网掩码，网关成功';
                res = {
                    status: '成功',
                    msg
                };
                return resolve(res);
            }
        );
    });
}
ipcMain.handle('change_net', async (event, netInfo) => {
    console.log('changeNetInfo', netInfo);

计算 ip 地址可选范围 ​

1.根据 cidr（192.163.2.234/23）计算 ip 范围 ​

js
// 根据cidr计算ip范围
function calculateIPRange(cidr) {
    console.log('cidr', cidr);
    function ipToInt(ip) {
        return ip.split('.').reduce((int, octet) => (int << 8) + parseInt(octet), 0) >>> 0;
    }
    function intToIp(int) {
        return [(int >>> 24) & 255, (int >>> 16) & 255, (int >>> 8) & 255, int & 255].join('.');
    }

    const [ip, maskbit] = cidr.split('/');
    const mask = -1 << ((32 - maskbit) >>> 0);
    const ipInt = ipToInt(ip);
    const netStartInt = (ipInt & mask) + 1;
    const netEndInt = (netStartInt | ~mask) - 1;

    let startNum = Number(intToIp(netStartInt).split('.').at(-1));
    let endNum = Number(intToIp(netEndInt).split('.').at(-1));

    // return {
    //     networkSegment: [intToIp(netStartInt), intToIp(netEndInt)].join(' - ')
    // };
    return { start: intToIp(netStartInt), end: intToIp(netEndInt), startNum, endNum };
}

// 根据ip地址和子网掩码计算ip范围
function calculateNetworkRange(address, netmask) {
    var ipArr = address.split('.').map((str) => parseInt(str));
    var subnetArr = netmask.split('.').map((str) => parseInt(str));
    var startIPArr = [],
        endIPArr = [];

    for (let i = 0; i < 4; i++) {
        startIPArr[i] = ipArr[i] & subnetArr[i];
        endIPArr[i] = startIPArr[i] | (~subnetArr[i] & 255);
    }

    startIPArr[3] += 1;
    endIPArr[3] -= 1;

    return {
        start: startIPArr.join('.'),
        end: endIPArr.join('.'),
        startNum: startIPArr[3],
        endNum: endIPArr[3]
    };
}

// 关机
function shutdown() {
    return new Promise((resolve, reject) => {
        let res = {};
        exec('shutdown -s -t 0', (error, stdout, stderr) => {
            if (error) {
                res = {
                    status: '失败',
                    msg: '关机失败'
                };
                return resolve(res);
            }

            let msg = '关机成功';
            res = {
                status: '成功',
                msg
            };
            return resolve(res);
        });
    });
}
ipcMain.handle('shutdown', async (event) => {
    let res = { ...(await shutdown()) };
    return res;
});
