问题： ​

1.this.canvas.set(“top”,200)更新 top 后元素不能无法再被选中 ​

当使用 Fabric.js 中的 set() 方法更新对象的位置属性（例如 left 和 top）时，该对象的位置会被更新并在 Canvas 上重新绘制，但是由于 Canvas 对象的选择控件实际上是通过浏览器的事件机制而不是画布上的 DOM 元素来实现的，因此在对象被移动后，选择控件可能需要重新定位才能操作该对象。

可以使用 setCoords() 方法强制更新对象的边界框并重新定位选择控件位置，例如：

js
// 将对象向右移动50个像素并强制刷新坐标
obj.set({ left: obj.left + 50 });
obj.setCoords();

            case 'triangle':

                    let left = this.getTransformedPosX(Math.min(this.mouseFrom.x, this.mouseTo.x));
                    let width = Math.abs(this.mouseFrom.x - this.mouseTo.x);
                    let height = Math.abs(this.mouseFrom.y - this.mouseTo.y);

                        width,
                        height,
                        top,
                        left

                    let top =
                        this.mouseTo.y > this.mouseFrom.y

                            : this.mouseFrom.y - ry * 2;

                        this.mouseTo.x > this.mouseFrom.x
                            ? this.mouseFrom.x
                            : this.mouseFrom.x - rx * 2;
                    canvasObject = Object.assign(canvasObject, {
                        rx,
                        ry,
                        top,
                        left
                    });
                }
                break;
            case 'rhomb':
                {
                    let x1 = Math.min(this.mouseFrom.x, this.mouseTo.x),
                        x2 = Math.max(this.mouseFrom.x, this.mouseTo.x),
                        y1 = Math.min(this.mouseFrom.y, this.mouseTo.y),

                }

                        y1 = this.getTransformedPosY(this.mouseFrom.y),

                    var degreesInRadians90 = (90 * PI) / 180;
                    var degreesInRadians270 = (270 * PI) / 180;

                    var degreesInRadians180 = (180 * PI) / 180;
                    var degreesInRadians180 = (180 * PI) / 180;

                    var dx = x2 - x1;
                    var dy = y2 - y1;
                    var angle = Math.atan2(dy, dx);
                    // var x225=p1.x+30*Math.cos(angle+degreesInRadians225);
                    // var y225=p1.y+30*Math.sin(angle+degreesInRadians225);

                    var y90 = y180 + arrowH * Math.sin(angle + degreesInRadians90);
                    var x270 = x180 + arrowH * Math.cos(angle + degreesInRadians270);
                    var y270 = y180 + arrowH * Math.sin(angle + degreesInRadians270);

                    let points = [
                        { x: x1, y: y1 },
                        { x: x180, y: y180 },
                        { x: x270, y: y270 },
                        { x: x2, y: y2 },

                        { x: x180, y: y180 },
                        { x: x1, y: y1 }
                    ];

                }
                break;
            case 'star':
                {
                    let x1 = Math.min(this.mouseFrom.x, this.mouseTo.x),
                        x2 = Math.max(this.mouseFrom.x, this.mouseTo.x),
                        y1 = Math.min(this.mouseFrom.y, this.mouseTo.y),

    // 更新元素属性
    setCanvasObject(data) {
        this.canvas.forEachObject((object) => {
            if (object.aID == data.aID) {
                // 图形
                if (!data.path && !data.points) {
                    object.set(data);
                }
                // 路径
                if (data.path) {
                    // 创建新的Path对象
                    let newPath = new fabric.Path(data.path);
                    let newPathObj = {};
                    Object.keys(newPath).map((item, index) => {
                        newPathObj[item] = newPath[item];
                    });
                    object.setOptions(newPathObj);
                }
                // 多边形
                if (data.points) {
                    // 创建新的Path对象
                    let newPath = new fabric.Polygon(data.points);
                    let newPathObj = {};
                    Object.keys(newPath).map((item, index) => {
                        newPathObj[item] = newPath[item];
                    });
                    object.setOptions(newPathObj);
                }
                object.setCoords();
            }
        });
        this.canvas.requestRenderAll();
    }

    // 绘制箭头的方法
    drawArrowFun(fromX, fromY, toX, toY, theta, headlen) {
        theta = typeof theta != 'undefined' ? theta : 30; // 箭头尖角度
        headlen = typeof theta != 'undefined' ? headlen : 10; // 箭头尖长度
        // 计算各角度和对应的P2,P3坐标
        var angle = (Math.atan2(fromY - toY, fromX - toX) * 180) / Math.PI,
            angle1 = ((angle + theta) * Math.PI) / 180,
            angle2 = ((angle - theta) * Math.PI) / 180,
            topX = headlen * Math.cos(angle1),
            topY = headlen * Math.sin(angle1),
            botX = headlen * Math.cos(angle2),
            botY = headlen * Math.sin(angle2);
        console.log('angle', angle, angle1, angle2);
        var arrowX = fromX - topX,
            arrowY = fromY - topY;
        var path = ' M ' + fromX + ' ' + fromY;
        path += ' L ' + toX + ' ' + toY;
        arrowX = toX + topX;
        arrowY = toY + topY;
        path += ' M ' + arrowX + ' ' + arrowY;
        path += ' L ' + toX + ' ' + toY;
        arrowX = toX + botX;
        arrowY = toY + botY;
        path += ' L ' + arrowX + ' ' + arrowY;
        return path;
    }

    drawStarFun() {
        var x1 = this.mouseFrom.x,
            x2 = this.mouseTo.x,
            y1 = this.mouseFrom.y,
            y2 = this.mouseTo.y;
        // let x1 = Math.min(this.mouseFrom.x, this.mouseTo.x),

        // y1 = Math.min(this.mouseFrom.y, this.mouseTo.y),
        // y2 = Math.max(this.mouseFrom.y, this.mouseTo.y);
        /**
         * 实现思路  (x1,y1)表示鼠标起始的位置 (x2,y2)表示鼠标抬起的位置
         * r 表示五边形外圈圆的半径，这里建议自己画个图理解
         * 正五边形夹角为36度。计算出cos18°，sin18°备用
         */
        var w = Math.abs(x2 - x1),
            h = Math.abs(y2 - y1),
            r = Math.sqrt(w * w + h * h);
        var cos18 = Math.cos((18 * Math.PI) / 180);
        var sin18 = Math.sin((18 * Math.PI) / 180);
        /**
         * 算出对应五个点的坐标转化为路径
         */
        var point1 = [x1, y1 + r];
        var point2 = [x1 + 2 * r * sin18, y1 + r - 2 * r * cos18];
        var point3 = [x1 - r * cos18, y1 + r * sin18];
        var point4 = [x1 + r * cos18, y1 + r * sin18];
        var point5 = [x1 - 2 * r * sin18, y1 + r - 2 * r * cos18];
        var path = ' M ' + point1[0] + ' ' + point1[1];
        path += ' L ' + point2[0] + ' ' + point2[1];
        path += ' L ' + point3[0] + ' ' + point3[1];
        path += ' L ' + point4[0] + ' ' + point4[1];
        path += ' L ' + point5[0] + ' ' + point5[1];
        path += ' Z';
        console.log('path', path);
        // let path = "M 642 388.175368329516 L 700.2035785306813 209.0431728809512 L 552.4339022757176 323.10178926534064 L 731.5660977242824 323.10178926534064 L 583.7964214693187 209.0431728809512"
        let canvasObject = new fabric.Path(path, {
            troke: this.drawColor,
            fill: 'rgba(255,255,255,1)',
            strokeWidth: this.lineWidth
            // angle:180,  //设置旋转角度
        });

        // 防止出现好多个
        if (canvasObject) {
            this.canvas.add(canvasObject);
            this.drawingObject = canvasObject;
        }
    }
