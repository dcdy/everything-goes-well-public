i18n 国际化翻译 ​

https://kazupon.github.io/vue-i18n/zh/introduction.html

安装 ​

shell
pnpm install vue-i18n
1

配置 ​

index.js

Vue2+ElementUI 版本 ​

js
// Vue2+ElementUI版本
// 中英文切换
import Vue from 'vue';
import VueI18n from 'vue-i18n';
import ElementLocal from 'element-ui/lib/locale';
//引入element-ui 语言包
import elementZH from 'element-ui/lib/locale/lang/zh-CN';
import elementEN from 'element-ui/lib/locale/lang/en';
//引入自己的语言包
const zh = require('./zh-CN.json');
const en = require('./en-GB.json');
Vue.use(VueI18n);

const i18n = new VueI18n({
    locale: localStorage.getItem('languageSet') || 'zh', // 语言标识
    legacy: false, // 使用 Composition API 模式，则需要将其设置为false
    globalInjection: true, // 全局模式，可以直接使用 $t
    messages: {
        // 'zh': zh,
        zh: Object.assign(zh, elementZH), // 合并中文语言包
        en: Object.assign(en, elementEN) // 合并英文语言包
    }
});
// 解析ElementUI语言包
ElementLocal.i18n((key, value) => i18n.t(key, value));
export default i18n;

Vue3+ElementPlus 版本 ​

https://element-plus.gitee.io/zh-CN/guide/i18n.html

js
import { createI18n } from 'vue-i18n';

import zh from './zh-CN.json';
import en from './en-GB.json';

// i18n
// 中英文切换
//引入自己的语言包
const i18n = createI18n({
    legacy: false, // 使用 Composition API 模式，则需要将其设置为false
    globalInjection: true, // 全局模式，可以直接使用 $t
    locale: localStorage.getItem('languageSet') || 'zh', // 语言标识
    messages: {
        zh: zh,
        en: en
    }
});

export default i18n;

vscode 配置 ​

安装 i18n Ally 插件 ​

配置 ​

.vscode/settings.json

json
"i18n-ally.localesPaths": ["src/utils/locales"], // 相对于项目根目录的语言环境目录路径
"i18n-ally.ignoredLocales": ["src/utils/locales"],// 忽略的语言环境文件
// "i18n-ally.pathMatcher": "{locale}/index.{ext}",
"i18n-ally.annotationInPlace": false, // 是否在代码中直接显示注释。如果设置为 true，将在代码中显示翻译。
"i18n-ally.translate.saveAsCandidates": true, // 是否将翻译保存为候选项
"i18n-ally.translate.engines": [
	"google-cn",
	"google",
	"deepl",
	"libretranslate"
], // 翻译引擎顺序
"i18n-ally.sourceLanguage": "zh-CN", // 设置源语言，其他语言的翻译将基于此语言。
"i18n-ally.displayLanguage": "zh-CN", // 设置显示语言
"i18n-ally.enabledParsers": ["js", "json", "ts"], // 翻译文件locales中的index文件的后缀可允许的格式，默认json
"i18n-ally.keystyle": "nested", // 设置翻译后的变量格式，"nested" 表示嵌套式，"flat" 表示扁平式。
"i18n-ally.sortKeys": true, // 是否对键进行排序
"i18n-ally.enabledFrameworks": ["vue", "react", "svelte", "general"],// 设置启用的框架
"i18n-ally.extract.keygenStyle": "camelCase", // 设置提取时的键生成风格，"camelCase" 表示驼峰式。
"i18n-ally.extract.keygenStrategy": "random", // 生成密钥路径的策略。可以slug，random或empty
"i18n-ally.keysInUse": [
    "view.progress_submenu.translated_keys",
    "view.progress_submenu.missing_keys",
    "view.progress_submenu.empty_keys"
],
