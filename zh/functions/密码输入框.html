html
<div class="login-main-input">
    <el-input
        ref="password"
        placeholder="请输入您的密码"
        v-model="password"
        :type="passwordType"
        maxlength="30"
    >
        <div
            slot="suffix"
            class="login-main-input-icon"
            @click="showPwd()"
        >
            <el-image
                :src="passwordType === 'password' ? $imageUrl + 'eye-close.png' : $imageUrl + 'eye-open.png'"
                fit="cover"
            ></el-image>
        </div>
    </el-input>
</div>

// 切换密码类型
showPwd() {
    if (this.passwordType === "password") {
        this.passwordType = "";
    } else {
        this.passwordType = "password";
    }
    this.$nextTick(() => {
        this.$refs.password.focus();
    });
},
