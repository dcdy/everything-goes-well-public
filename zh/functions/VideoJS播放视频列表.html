// 定义变量
let playPC = null; // 播放的 RTCPeerConnection 对象

// 开始播放流
async function startPlay(streamPath) {
    const dom = document.getElementById(domId);
    playPC = new RTCPeerConnection();

    // 处理 ontrack 事件
    playPC.ontrack = (e) => {
        console.log('ontrack', e);
        if (e.streams.length === 0 || !dom) return;
        dom.srcObject = e.streams[0];
    };

    // 处理连接状态变化
    playPC.oniceconnectionstatechange = (e) => {
        console.log('oniceconnectionstatechange', e, playPC, playPC.iceConnectionState);
        if (playPC.iceConnectionState === 'connected') {
            console.log('播放成功', streamPath);
            // emit('startPlayEmit', streamPath);
        }

        if (['disconnected', 'failed', 'closed'].includes(playPC.iceConnectionState)) {
            console.log('播放结束', streamPath);
            stopPlay(streamPath); // 停止播放
        }
    };

    playPC.onconnectionstatechange = (e) => {
        console.log('state changed', e, playPC);
    };

    playPC.onicecandidate = (e) => {
        console.log('onicecandidate', e);
    };

    // 添加接收器
    playPC.addTransceiver('video', { direction: 'recvonly' });
    playPC.addTransceiver('audio', { direction: 'recvonly' });

    // 创建 offer 并设置本地描述
    const offer = await playPC.createOffer();
    await playPC.setLocalDescription(offer);

    // 发送 SDP 到服务器
    const result = await fetch(streamPath, {
        method: 'POST',
        mode: 'cors',
        cache: 'no-cache',
        credentials: 'include',
        redirect: 'follow',
        referrerPolicy: 'no-referrer',
        headers: { 'Content-Type': 'application/sdp' },
        body: offer.sdp
    }).catch((err) => {
        console.error('播放失败', err);
        // emit('stopPlayEmit', streamPath);
    });

    if (!result || result.status !== 200) {
        // emit('stopPlayEmit', streamPath);
        return;
    }

    const remoteSdp = await result.text();
    await playPC.setRemoteDescription(
        new RTCSessionDescription({ type: 'answer', sdp: remoteSdp })
    );

    console.log('播放成功', streamPath);
}
// 停止播放
function stopPlay(streamPath) {
    console.log('结束播放流', streamPath, playPC);
    const dom = document.getElementById(domId);

    if (dom) {
        stopTrack(dom.srcObject);
    }

    if (playPC) {
        playPC.close();
        playPC = null;
    }

    // emit('stopPlayEmit', streamPath);
}
// 停止播放流
function stopTrack(stream) {
    if (stream) {
        const tracks = stream.getTracks();
        tracks.forEach((track) => track.stop());
    }
}

        });

    }
