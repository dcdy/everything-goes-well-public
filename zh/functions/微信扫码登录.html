3.注：如果想获取扫码过程中的状态，例如扫描成功、用户电力拒绝登录 ​

可以在 network 中查看下地址中的状态码 window.wx_errcode=404

405：登录成功 404：扫描成功 403：取消登录 408：需要再次请求 500：重新获取 uuid

2.网站用 ​

如果是网站用直接将 redirect_uri 的重定向地址指向自己扫码登录成功后要跳转的地址就可以，但必须是线上地址

3.electron ​

如果是网站 electron 用，需要先拦截重定向要跳转的地址，在当前地址获取 code

参考：https://blog.csdn.net/qq_21440443/article/details/120196889

使用 sesson 的 webRequest.onBeforeRequest 方法拦截

https://www.electronjs.org/zh/docs/latest/api/web-request#webrequestonbeforesendheadersfilter-listener

主进程内的 app-ready 内使用

js
// 拦截微信登录成功后要重定向的地址，方便获取code
// filter是需要配置的拦截规则
// 加?code=时防止正式接口时是www.jing.top都给拦截了
const filter = { urls: ['*://www.jing.top/?code=*'] };
// win 当前窗体
session.defaultSession.webRequest.onBeforeSendHeaders(filter, (details, next) => {
    console.log('捕获url规则内的请求', details.url);
    // details.requestHeaders['Referer'] = 'http://www.jing.top'
    // callback({ requestHeaders: details.requestHeaders });
    if (details.url.indexOf('www.jing.top/?code=') != -1) {
        // 发送信息到渲染进程
        win.webContents.send('main-process-webRequest', details.url);
    }
    // 执行next将放开拦截
    // next({ cancel: false, requestHeaders: details.requestHeaders});
});

// 获取微信登录成功后的code
getWxLoginCode(){
    window.electronAPI.mainProcessWebRequest((e, url) => {
        console.log("//////", url);
        this.wxCode = getQueryObject(url).code
        console.log('this.wxCode',this.wxCode);
    });
},

4.完整功能 ​

vue
<template>
    <div class="wrap">
        <LoginBg></LoginBg>
        <div class="container flex-center">
            <div class="login">
                <div class="login-head">
                    <div class="login-head-left">
                        <div class="login-head-title">微信扫码，安全登录</div>
                        <div class="login-head-subtitle">【精勤直播】</div>
                    </div>
                    <div class="login-head-right">
                        <el-image
                            style="cursor: pointer;"
                            :src="$imageUrl + 'PC.png'"
                            fit="cover"
                            @click="stepLoginPage()"
                        ></el-image>
                        <div class="login-head-code"></div>
                    </div>
                </div>
                <div class="login-main">
                    <wxlogin
                        v-if="appid && redirect_uri"
                        :appid="appid"
                        scope="snsapi_login"
                        :redirect_uri="redirect_uri"
                        :href="href"
                        :state="state"
                    ></wxlogin>
                </div>
                <div class="login-main-set">
                    <el-checkbox v-model="isAutoLoginCheck">保持自动登录</el-checkbox>
                </div>
            </div>
        </div>
    </div>
</template>

<script>
import LoginBg from '@/components/LoginBg.vue';
// 接入微信扫码登录组件
import wxlogin from 'vue-wxlogin';
import { getQueryObject } from '@/utils/index.js';
import { getWxCode } from '@/api/loginApi.js';
import { setCookie, isExpiresFun } from '@/utils/tools.js';
export default {
    name: 'WxLogin',
    components: {
        LoginBg,
        wxlogin
    },
    data() {
        return {
            account: '', //账号
            password: '', //密码
            passwordType: 'password', //密码类型是否可见
            isAutoLoginCheck: false, //是否自动登录

            // 微信扫码登陆参数
            appid: 'wx9acc8926d5605cba', // 微信开放平台内的appid
            redirect_uri: encodeURIComponent('https://www.jingqin.top'), // 微信开放平台内配置的重定向地址
            state: '1', //
            href: 'data:text/css;base64,LmltcG93ZXJCb3ggLnFyY29kZSB7d2lkdGg6IDE4MHB4O21hcmlnbi10b3A6MHB4ICFpbXBvcnRhbnQ7Ym9yZGVyOiBub25lO30KLmltcG93ZXJCb3ggLnRpdGxlIHtkaXNwbGF5OiBub25lO30KLmltcG93ZXJCb3ggLmluZm8ge2Rpc3BsYXk6IG5vbmU7fQ==', // 自定义样式链接

            wxCode: '', // 微信登录成功后获取的code值
            wxLoginCount: 0 // 记录扫码登录请求次数
        };
    },
    watch: {
        isAutoLoginCheck(newVal, oldVal) {
            // 缓存是否自动登录
            localStorage.setItem('isAutoLogin', newVal);
        },
        wxCode(newVal, oldVal) {
            // 不知道什么原因多次扫码会多次请求
            // 但因为code都是不变的，所以用watch可以保证只请求一次
            // 调用接口获取信息
            getWxCode({ code: this.wxCode, state: 1 }).then((res) => {
                this.$loading().close();
                console.log('res', res);
                if (res.code == 200) {
                    // 如果还没注册过，就跳到注册页面
                    if (res.data.register) {
                        this.$router.push('Register');
                    } else {
                        // 如果已经注册过了，直接跳到首页
                        this.$router.push('Login');
                    }
                    // 如果选择了自动登录，就cookie中存储自动登录信息和时间，7是7天
                    if (this.isAutoLoginCheck) {
                        setCookie('isWxLogin', 'true', 7);
                    }
                } else {
                    this.$message.error(
                        'code已经被其他设备使用或code已过期，请点击二维码刷新后重新扫码登录'
                    );
                }
            });
        }
    },
    created() {
        console.log('createdcreatedcreated');
        // 获取缓存中是否自动登录的值
        this.isAutoLoginCheck = JSON.parse(localStorage.getItem('isAutoLogin'));
        // 记录扫码登录请求次数
        this.wxLoginCount = 0;
    },
    mounted() {
        console.log('mountedmountedmounted');
        // 获取微信登录成功后的code
        this.getWxLoginCode();

        // 如果已经登录过了
        // 判断是否过期
        if (!isExpiresFun('isWxLogin')) {
            this.$loading();
            // 判断是否登录
            setTimeout(() => {
                this.$loading().close();
                // this.loginFun();
                this.$router.push('Login');
            }, 2000);
        }
    },
    methods: {
        // 跳转到登录页
        stepLoginPage() {
            this.$router.push('Login');
        },
        // 更新二维码
        resetWxCode() {
            this.appid = '';
            this.wxCode = '';
            setTimeout(() => {
                this.appid = 'wx9acc8926d5605cba';
            }, 500);
        },
        // 获取微信登录成功后的code
        getWxLoginCode() {
            window.electronAPI.mainProcessWebRequest((e, url) => {
                this.$loading();
                // 获取code
                this.wxCode = getQueryObject(url).code;
            });
        }
    }
};
</script>
