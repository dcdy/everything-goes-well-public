export interface Options {
  /**
   * 渲染前应用于节点的宽度（像素）。
   */
  width?: number

  /**
   * 渲染前应用于节点的高度（像素）。
   */
  height?: number

  /**
   * 表示 JPEG 图像质量的数字，范围在 `0` 到 `1` 之间（例如，0.92 表示 92% 的质量）。
   */
  quality?: number

  /**
   * 表示图像格式的字符串。默认类型为 image/png；如果给定的类型不支持，则也使用该类型。
   */
  type?: string

  /**
   * 捕获图像的像素比率。
   *
   * DPI = 96 * scale
   *
   * 默认值：1
   */
  scale?: number

  /**
   * 表示背景颜色的字符串值，可以是任何有效的 CSS 颜色值。
   */
  backgroundColor?: string | null

  /**
   * 在渲染前复制到节点样式的对象。
   */
  style?: Partial<CSSStyleDeclaration> | null

  /**
   * 接受 DOM 节点作为参数的函数。如果返回 `true`，则传递的节点应包含在输出中。
   * 排除节点意味着排除其子节点。
   */
  filter?: ((el: Node) => boolean) | null

  /**
   * 最大画布尺寸（像素）。
   *
   * 参考：https://developer.mozilla.org/en-US/docs/Web/HTML/Element/canvas#maximum_canvas_size
   */
  maximumCanvasSize?: number

  /**
   * 加载媒体和获取远程资产的超时时间（毫秒）。
   *
   * 默认值：30000
   */
  timeout?: number

  /**
   * 嵌入资产的进度。
   */
  progress?: ((current: number, total: number) => void) | null

  /**
   * 启用调试模式以查看执行时间日志。
   */
  debug?: boolean

  /**
   * 获取自定义 URL 的图像数据的自定义实现。
   * 这对于使用 Capacitor 或 Cordova 时使用原生 fetch 绕过 CORS 问题非常有帮助。
   *
   * 如果返回一个字符串，则完全绕过任何 `Options.fetch` 设置并使用你的自定义实现。
   *
   * 如果返回 false，则回退到正常的 fetch 实现
   *
   * @param url
   * @returns 图片的数据 URL
   */
  fetchFn?: ((
    url: string,
  ) => Promise<string | false>) | null

  /**
   * 获取资源的选项。
   */
  fetch?: {
    /**
     * `window.fetch` 的第二个参数 RequestInit
     *
     * 默认值：{
     *   cache: 'force-cache',
     * }
     */
    requestInit?: RequestInit

    /**
     * 设置为 `true` 时，向 URL 请求追加当前时间作为查询字符串以启用缓存清除。
     *
     * 默认值：false
     */
    bypassingCache?: boolean | RegExp

    /**
     * 用于占位的图像数据 URL，当获取图像失败时使用。
     * 默认值为空字符串，并将在失败的图像区域渲染空白区域。
     *
     * 默认值：data:image/png;base64,R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7
     */
    placeholderImage?: string | ((cloned: HTMLImageElement | SVGImageElement) => string | Promise<string>)
  }

  /**
   * 字体下载和嵌入的选项。
   */
  font?: false | {
    /**
     * 首选的字体格式。如果指定，则其他所有字体格式将被忽略。
     */
    preferredFormat?: 'woff' | 'woff2' | 'truetype' | 'opentype' | 'embedded-opentype' | 'svg' | string

    /**
     * 用于指定字体嵌入的 CSS 字符串。如果指定，则仅此 CSS 会出现在生成的图像中。
     */
    cssText?: string
  }

  /**
   * 所有启用的功能
   *
   * 默认值：true
   */
  features?: boolean | {
    /**
     * 复制滚动条 CSS 样式
     *
     * 默认值：true
     */
    copyScrollbar?: boolean

    /**
     * 删除克隆节点上的异常属性（用于规范化 XML）
     *
     * 默认值：true
     */
    removeAbnormalAttributes?: boolean

    /**
     * 删除控制字符（用于规范化 XML）
     *
     * 默认值：true
     */
    removeControlCharacter?: boolean

    /**
     * 修复 svg+xml 图像解码（针对 Safari、Firefox）
     *
     * 默认值：true
     */
    fixSvgXmlDecode?: boolean

    /**
     * 使用滚动内容渲染滚动的子节点
     *
     * 默认值：false
     */
    restoreScrollPosition?: boolean
  }

  /**
   * Canvas `drawImage` 的间隔
   * 用于修复 Safari、Firefox 中的图像解码错误
   *
   * 默认值：100
   */
  drawImageInterval?: number

  /**
   * Web Worker 脚本 URL
   */
  workerUrl?: string | null

  /**
   * Web Worker 的数量
   */
  workerNumber?: number

  /**
   * 节点克隆后触发
   */
  onCloneNode?: ((cloned: Node) => void) | null

  /**
   * 节点嵌入后触发
   */
  onEmbedNode?: ((cloned: Node) => void) | null

  /**
   * ForeignObjectSvg 创建后触发
   */
  onCreateForeignObjectSvg?: ((svg: SVGSVGElement) => void) | null

  /**
   * 样式属性名的数组。
   * 可用于手动指定克隆节点时包含的样式属性。
   * 这对于性能关键的场景非常有用。
   */
  includeStyleProperties?: string[] | null
}

js
import { domToPng } from 'modern-screenshot';
const generateImage = async () => {
    // 获取要生成图片的dom节点
    const node = document.getElementById('chart-id');
    if (!node) {
        return;
    }
    try {
        const dataUrl = await domToPng(node, {
            // 传入配置
            scale: 2,
            timeout: 10000
        });

        // 通过a标签自动下载图片
        const a = document.createElement('a');
        a.href = dataUrl;
        a.download = 'chart.png';
        a.click();
        a.remove();
        ElMessage.success('图片生成成功，请耐心等待下载');
    } catch (error) {
        ElMessage.error('图片生成失败');
    }
};
