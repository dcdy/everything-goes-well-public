基于 faceapi.js 库 ​

html
<!-- 捕获来自用户媒体设备的视频流 -->
<div style="position: relative">
    <video
        ref="videoElement"
        playsinline
        autoplay
        muted
        @loadedmetadata="onPlay()"
    ></video>
    <!-- 在这里显示 canvas，脸部检测的结果会被画在这个 canvas 上 -->
    <canvas
        style="position: absolute; left: 0; top: 0"
        ref="faceCanvas"
    ></canvas>
</div>

vue
<script setup>
// 导入 face-api.js 库
import * as faceapi from 'face-api.js';
const emit = defineEmits(['getFaceDetectResult']);

// 获取容器dom
let videoElement = ref(null);
let faceCanvas = ref(null);
let canvasContext = null;
// 模型检测器类型
let faceDetectionNet = 'tiny_face_detector'; // 'ssd_mobilenetv1' | 'tiny_face_detector'
// 初始化
async function init() {
    // 负荷面检测模型
    await changeFaceDetector(faceDetectionNet);
    // 获取摄像头流
    await getVideoStream(); // 获取 canvas 上下文
    canvasContext = faceCanvas.value.getContext('2d');
}
// 获取摄像头流
async function getVideoStream() {
    const stream = await navigator.mediaDevices.getUserMedia({ video: {} });
    videoElement.value.srcObject = stream;
}

let faceDetectionOptions = null; // 面部检测器选项
let drawOptions = null; // 人脸检测框样式内容
// 播放视频
async function onPlay() {
    if (videoElement.value.paused || videoElement.value.ended || !isFaceDetectionModelLoaded())
        return setTimeout(() => onPlay());
    // 获取面部检测器选项
    faceDetectionOptions = getFaceDetectorOptions('tiny_face_detector');

    // 计算 fps
    // const ts = Date.now();

    // 获取面部检测结果
    const result = await getFaceDetectionResult('2', videoElement.value);
    console.log('result', result, videoElement.value.paused);

    // 计算 fps
    // updateTimeStats(Date.now() - ts);
    if (result) {
        // canvas 绘制人脸检测框
        drawOptions = {
            label: '检测中，请保持正脸',
            lineWidth: 2,
            boxColor: 'red'
        };
        drawFaceDetection(result, faceCanvas.value, videoElement.value, drawOptions);

        // 计算距离
        const leftEye = result.landmarks.getLeftEye();
        const rightEye = result.landmarks.getRightEye();
        // 判断左右眼睛是否在同一水平线上(判断正脸)
        const diffY = Math.abs(leftEye[0].y - rightEye[0].y);
        console.log('左右眼睛距离', diffY);
        // setTimeout(() => {
        if (diffY <= 5 && result.detection.score > 0.85) {
            // 停止人脸识别
            if (videoElement.value) {
                videoElement.value.pause();
                canvasContext.clearRect(0, 0, faceCanvas.value.width, faceCanvas.value.height);
                // canvas 绘制人脸检测框
                drawOptions = {
                    label: '检测成功',
                    lineWidth: 2,
                    boxColor: 'green'
                };
                drawFaceDetection(result, faceCanvas.value, videoElement.value, drawOptions);
                emit('getFaceDetectResult', result);
            }
        }
        // }, 1000);
    } else {
        // 清除旧的绘制框
        if (videoElement.value.paused || videoElement.value.ended) return;
        canvasContext.clearRect(0, 0, faceCanvas.value.width, faceCanvas.value.height);
    }

    setTimeout(() => onPlay());
}
// 获取当前面部检测器
function getCurrentFaceDetectionNet(selectedFaceDetector) {
    console.log('获取当前面部检测器', selectedFaceDetector);
    if (selectedFaceDetector === 'ssd_mobilenetv1') {
        return faceapi.nets.ssdMobilenetv1;
    }
    if (selectedFaceDetector === 'tiny_face_detector') {
        return faceapi.nets.tinyFaceDetector;
    }
}
// 是否加载了面部检测模型
function isFaceDetectionModelLoaded() {
    return !!getCurrentFaceDetectionNet(faceDetectionNet).params;
}
// 更改面部检测器
async function changeFaceDetector(detector) {
    console.log('changeFaceDetector', isFaceDetectionModelLoaded, detector);
    if (!isFaceDetectionModelLoaded()) {
        // 加载模型
        let path = new URL('/models', import.meta.url).pathname.replace('/@fs', '');
        await getCurrentFaceDetectionNet(faceDetectionNet).loadFromUri(path);
        await faceapi.nets.faceRecognitionNet.loadFromUri(path);
        // await faceapi.nets.faceLandmark68Net.loadFromUri('path');
        await faceapi.nets.faceLandmark68TinyNet.loadFromUri(path);
        // 加上面部表情识别👇
        // await faceapi.nets.faceExpressionNet.loadFromUri('path');
        // await faceapi.nets.ssdMobilenetv1.loadFromUri('path')
        // 加上面部表情识别👆
    }
}
// 获取面部检测器选项
function getFaceDetectorOptions(selectedFaceDetector) {
    return selectedFaceDetector === 'ssd_mobilenetv1'
        ? new faceapi.SsdMobilenetv1Options({ minConfidence: 0.5 })
        : new faceapi.TinyFaceDetectorOptions({ inputSize: 128, scoreThreshold: 0.5 });
}

/**
 * 获取面部检测结果(可以任意组合，但要加载相应模型)
 * detectSingleFace: 检测单个人脸
 * detectAllFaces: 检测所有人脸
 * withFaceLandmarks: 检测68个面部标志点
 * withFaceDescriptor: 检测人脸描述
 * withFaceExpressions: 检测面部表情
 * withAgeAndGender: 检测年龄和性别
 * @param {string} type - 检测类型
 * @param {HTMLElement} ele - 检测的元素dom
 * @returns {Promise} - 返回检测结果
 */
async function getFaceDetectionResult(type = '2', ele) {
    let map = new Map([
        ['0', faceapi.detectSingleFace(ele, faceDetectionOptions)],
        ['1', faceapi.detectSingleFace(ele, faceDetectionOptions).withFaceLandmarks(true)],
        [
            '2',
            faceapi
                .detectSingleFace(ele, faceDetectionOptions)
                .withFaceLandmarks(true)
                .withFaceDescriptor()
        ],
        [
            '3',
            faceapi
                .detectSingleFace(ele, faceDetectionOptions)
                .withFaceLandmarks(true)
                .withFaceDescriptor()
                .withFaceExpressions()
        ],
        ['4', faceapi.detectSingleFace(ele, faceDetectionOptions).withAgeAndGender()],
        ['00', faceapi.detectAllFaces(ele, faceDetectionOptions)],
        ['11', faceapi.detectAllFaces(ele, faceDetectionOptions).withFaceLandmarks(true)],
        [
            '22',
            faceapi
                .detectAllFaces(ele, faceDetectionOptions)
                .withFaceLandmarks(true)
                .withFaceDescriptors()
        ],
        [
            '33',
            faceapi
                .detectAllFaces(ele, faceDetectionOptions)
                .withFaceLandmarks(true)
                .withFaceDescriptors()
                .withFaceExpressions()
        ],
        ['44', faceapi.detectAllFaces(ele, faceDetectionOptions).withAgeAndGender()]
    ]);
    const result = await map.get(type); // 使用 await 关键字等待 Promise 完成
    return result;
}

// 用于计算 fps
let forwardTimes = [];
let timeStats = ref({
    time: '',
    fps: ''
});
function updateTimeStats(timeInMs) {
    forwardTimes = [timeInMs].concat(forwardTimes).slice(0, 30);
    const avgTimeInMs = forwardTimes.reduce((total, t) => total + t) / forwardTimes.length;
    timeStats.value = {
        time: `${Math.round(avgTimeInMs)} ms`,
        fps: `${faceapi.utils.round(1000 / avgTimeInMs)}`
    };
}

// canvas 绘制人脸检测框
function drawFaceDetection(result, canvasDom, eleDom, drawOptions) {
    const dims = faceapi.matchDimensions(canvasDom, eleDom, true);
    if (!drawOptions) {
        // 默认的人脸检测框
        let size = faceapi.resizeResults(result, dims);
        // 默认的蓝色框加上匹配分数
        faceapi.draw.drawDetections(canvasDom, faceapi.resizeResults(result, dims));
        // 默认的面部标志
        faceapi.draw.drawFaceLandmarks(canvasDom, faceapi.resizeResults(result, dims));
    } else {
        // 修改颜色和文字的人脸检测框
        const drawBox = new faceapi.draw.DrawBox(result.detection.box, drawOptions);
        drawBox.draw(canvasDom);
    }
}
onMounted(async () => {
    init();
});
onUnmounted(() => {
    // 清理
    if (videoElement.value) {
        const stream = videoElement.value.srcObject;
        const tracks = stream.getTracks();
        tracks.forEach((track) => {
            track.stop();
        });
        videoElement.value.srcObject = null;
    }
});
</script>
