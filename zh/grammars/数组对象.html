**splice()：**数组中增删改 ​

  • 从数组中删除任意个（可以是负的即增加）元素。
  • 返回结果为被删除元素组成的新数组。
  • 该方法会改变原数组，会将指定元素从原数组中删除；

语法：

js
// 新数组 = 原数组.splice(起始索引index, 需要删除的个数);

// 新数组 = 原数组.splice(起始索引index, 需要删除的个数, 新的元素1, 新的元素2...);

var arr1 = ['a', 'b', 'c', 'd', 'e', 'f'];
arr1.splice(1); //从第index为1的位置开始，删除元素
arr2.splice(-2); //删除最后两个元素，和slice同样的思想。
arr3.splice(1, 3); //从第index为1的位置开始删除元素,一共删除三个元素
// 增加系列
arr4.splice(1, 0, 'g', 'h'); //纯增加情况
//变更的情况就是 先删除再增加，即替换
arr4.splice(1, 3, 'js', 'vue'); //删除+增加 == 更改

slice()：截取字符串 ​

  • 从数组中截取任意个元素，返回结果为新的数组
  • 不改变原数组。

语法：

js
新数组 = 原数组.slice(开始位置的索引, 结束位置的索引);
//注意：包含开始索引，不包含结束索引，即[a,b)
举例：
const arr = ['a', 'b', 'c', 'd', 'e', 'f'];
arr.slice();
// 无参数时，截取所有的元素。
arr.slice(2);
// 从第二个值开始提取，直到末尾
arr.slice(-2);
// 提取最后两个元素 负数会将其与长度相加，
-2+6=4 arr.slice(2, 4);
// 提取从第二个到第四个之间的元素（不包括第四个元素）
arr.slice(4, 2);
// 空

5.两个数组对象替换属性 ​

（1）筛选 arr1 中 id 和 arr2 一样值的数据，将 arr1 中对象全部属性替换为 arr2 对应的对象属性，未找到匹配对象的元素会保留原来 arr1 的对象 ​

js
function replaceObjects(arr1, arr2, property) {
    const map = new Map();
    // 将 arr2 中的对象以 property 值为 key 存入 map
    arr2.forEach((obj) => {
        const key = obj[property];
        map.set(key, obj);
    });
    // 遍历 arr1，替换相同 property 值的对象，不存在的保留原对象
    arr1.forEach((obj, index) => {
        const key = obj[property];
        if (map.has(key)) {
            arr1[index] = map.get(key);
        }
    });

    return arr1;
}
// 两个测试数组
const arr1 = [
    { id: 1, name: 'Alice', age: 18 },
    { id: 2, name: 'Bob', age: 28 },
    { id: 3, name: 'Charlie', age: 38 }
];
const arr2 = [
    { id: 2, name: 'Bobby' },
    { id: 3, name: 'Charles', age: 48, sex: '女' },
    { id: 4, name: 'Dave' }
];
// 测试调用
const result = replaceObjects(arr1, arr2, 'id');
console.log(result);
// 结果

    { id: 2, name: 'Bobby' },

function replaceObjects(arr1, arr2, property) {
    const map = new Map();
    // 将 arr1 中的对象以 property 值为 key 存入 map
    arr1.forEach((obj) => {
        const key = obj[property];
        map.set(key, obj);
    });
    // 遍历 arr2，替换 arr1 中相同 property 值的对象，保留其他属性

（4）以 arr2 为主，根据 arr2 的长度来调整 arr1 的长度 ​

js
function replaceObjects(arr1, arr2, property) {
    const map = new Map();
    // 将 arr2 中的对象以 property 值为 key 存入 map
    arr2.forEach((obj) => {
        const key = obj[property];
        map.set(key, obj);
    });
    // 遍历 arr1，替换相同 property 值的对象，保留其他属性
    for (let i = 0; i < arr1.length; i++) {
        const obj = arr1[i];
        const key = obj[property];
        if (map.has(key)) {
            const replaceObj = map.get(key);
            Object.assign(obj, replaceObj);
        } else {
            arr1.splice(i, 1);
            i--;
        }
    }
    // 将 arr2 中不存在于 arr1 的对象添加到 arr1 中
    for (let i = arr1.length; i < arr2.length; i++) {
        arr1.push(arr2[i]);
    }
    return arr1;
}
// 两个测试数组
const arr1 = [

    { id: 2, name: 'Bobby' },
    { id: 3, name: 'Charles', age: 48, sex: '女' },
    { id: 4, name: 'Dave' }
];
// 测试调用
const result = replaceObjects(arr1, arr2, 'id');
console.log(result);
// 结果
[
    { id: 2, name: 'Bobby' },

    { id: 4, name: 'Dave' }

示例 3：

定义一个 js 对象转数组的函数

js
function objtoarray(obj) {
    var array = [];
    for (var key in obj) {
        //判断一下是否为对象
        if (Object.prototype.toString.call(obj[key]) == '[object Object]') {
            array[key] = objtoarray(obj[key]);
        } else {
            array[key] = obj[key];
        }
    }
    return array;
}
