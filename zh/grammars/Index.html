a||b (a逻辑或b) ​

  • 1.只要 a 为真，不管后面 b 的值是多少，结果返回都是 a
js
// 1||2打印值为1
console.log(1 || 2);

// true||2打印值为true
console.log(true || 2);

原生 js 实现

https://www.cnblogs.com/zwb4461/p/11814204.html

insertAdjacentHTML() 将指定的文本解析为 HTML 或 XML，并将结果节点插入到 DOM 树中的指定位置。它不会重新解析它正在使用的元素，因此它不会破坏元素内的现有元素。这避免了额外的序列化步骤，使其比直接 innerHTML 操作更快。

js
element.insertAdjacentHTML(position, text);
1

position 有四个参数选项：

1.beforebegin:规定在目标元素的外部开始位置插入。

2.afterbegin:规定在目标元素的内部开始位置插入。

3.beforeend:规定在目标元素的内部结束位置插入。

4.afterend:规定在目标元素的外部结束位置插入。

js
let str = `<div class="dialog-chat-bantalk">房间已开启全体禁言！</div>`;
let dom = document.getElementById('dialog-chat');
// 向dialog-chat类的内部的最后处添加str内容
dom.insertAdjacentHTML('beforeend', str);

setInterval(() => {
    this.originArr = [];
    // 让渲染数组的值等于更新数组的值
    this.activeArr.map((e, i) => {
        this.originArr[i] = e;

        // 默认isV都给true

            if (e.id == e2.id) {
                console.log('等于不');
                e.isV = e2.isV;

        });
    });

    this.$forceUpdate();
}, 5000);

    let index = this.originArr.findIndex(e => e.id === id)
    // 改变isV值取反
    this.originArr[index].isV = !this.originArr[index].isV

    // 只要有点击过就在改值数组中push值
    this.newArr.push({id:id,isV:this.originArr[index].isV})
    // 改值数组去重，reduceRight代表后覆盖前，reduce代表前覆盖后
    let obj = {};
    this.newArr = this.newArr.reduceRight((cur,next) => {
        obj[next.id] ? "" : obj[next.id] = true && cur.push(next);
        return cur;
    },[]) //设置cur默认类型为数组，并且初始值为空的数组

    console.log("id",this.originArr,this.newArr);
    this.$forceUpdate()
},
